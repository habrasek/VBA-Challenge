Sub looper()For Each ws In Worksheets    ws.Cells(1, 12).Value = "Ticker"    ws.Cells(1, 13).Value = "Change"    ws.Cells(1, 14).Value = "Percent Change"    ws.Cells(1, 15).Value = "Total Volume"            Dim change As Double    Dim rowcount1 As Double    Dim summarycounter As Integer    Dim rowscount As Double        summarycounter = 2        rowscount = ws.Rows.Count - 1    rowcount1 = 2        For i = 2 To rowscount        If ws.Cells(i, 1).Value <> ws.Cells(i + 1, 1).Value Then            ws.Cells(summarycounter, 12).Value = ws.Cells(i, 1).Value            ws.Cells(summarycounter, 13).Value = ws.Cells(i, 6).Value - Cells(rowcount1, 3)            ws.Cells(summarycounter, 14).Value = Format((ws.Cells(i, 6).Value / ws.Cells(rowcount1, 3) - 1), "Percent")            ws.Cells(summarycounter, 15).Value = Application.Sum(Range(ws.Cells(rowscount, 7), ws.Cells(i, 7)))            If ws.Cells(summarycounter, 14).Value > 0 Then                ws.Cells(summarycounter, 14).Interior.ColorIndex = 4            End If            If ws.Cells(summarycounter, 14).Value < 0 Then                ws.Cells(summarycounter, 14).Interior.ColorIndex = 3            End If            summarycounter = summarycounter + 1        End If                                'If Cells(i, 1).Value <> Cells(i - 1, 1).Value Then        '    rowcount1 = i            'Cells(summarycounter, 14).Value = rowcount1                'End If    Next iNext wsEnd Sub
